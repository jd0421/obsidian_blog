https://solvesql.com/problems/sales-summary/


```mysql
with step1 as (

  SELECT

    day

    , sum(total_bill) as total_bill  

  from tips

  group by 1

)

  

select

  round(avg(total_bill), 2) as avg_sales

from step1

;
```
